           /*
           In Factory pattern, we create object without exposing the creation logic to the client            and refer to newly created object using a common interface.
           */
            if(order == null){
                return null;
            }
            
            if(order.equals("cake")){
                return new Cake();
            }

            if(order.equals("pizza")){
                return new Pizza();
            }
            
            return null;
